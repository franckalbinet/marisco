ReshapeLongToWide

 ReshapeLongToWide (columns='nuclide', values=['value'])

Convert data from long to wide with renamed columns.

# How do that work?
#    In our case, unique idx is a composite index (lon, lat, time, depth)
#    We want to transform:
# idx , nuc, val
# 1   , a  , 1
# 1   , b  , 2
# 2   , c  , 3

# to: 
# idx, a,  b,   c
# 1  , 1,  2,   nan
# 2  ,nan, nan, 3 
df_test = pd.DataFrame({
    'idx': [1, 1, 2],
    'nuclide': ['a', 'b', 'c'],
    'value': [1, 2, 3]
    })

df_test.pivot_table(index='idx',
                    columns='nuclide',
                    values='value',
                    fill_value=np.nan,
                    # aggfunc=lambda x: x
                    )
